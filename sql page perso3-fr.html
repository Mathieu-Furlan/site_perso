Le code lié au résultat ci-dessus:

SELECT IIF(a.ID IS NULL, (SELECT TOP 1 b.ID FROM donnee b WHERE b.Grp = a.Grp AND b.ID IS NOT NULL AND b.Month > a.Month), a.ID), a.Grp, a.Month
FROM donnee AS a
ORDER BY Grp, Month DESC;
