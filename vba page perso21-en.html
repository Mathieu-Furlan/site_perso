Twenty-first case: Stripping uppercase words.

Summary of the problem:

I want to exctract all words in uppercase and generate a list with them.

How to solve the problem:

	
	Sub touvermaj()

	Dim arr() As String
	Dim x As Integer
	Dim i As Integer
	Dim n As Integer
	Dim q As Integer
	Dim apos As String
	Dim newapos As String

	x = 1

	For i = 1 To Cells(Rows.Count, 1).End(xlUp).Row
		apos = Cells(i, 2)
		newapos = Replace(apos, "’", " ")
		arr = Split(newapos, " ")
		For n = 0 To UBound(arr)
			If arr(n) = UCase(arr(n)) And Not IsNumeric(arr(n)) Then
				Cells(x, 5) = arr(n)
				Cells(x, 6) = i
				x = x + 1
           
			End If
        
			Debug.Print arr(n)
		Next n
	Next i
    
	For q = 0 To UBound(arr)

		Debug.Print arr(q)

	Next q

	End Sub
