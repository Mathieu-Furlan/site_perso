Sixth case: highlight cells based on condition.

Summary of the problem:

I'm trying to loop through all values in column "D", and if a value ends with an "A", highlight it.

The code to obtain desired result:

	
	Sub dern()

	Dim x, y, z, i As Integer
	Dim a, b, c As String

	Randomize

	Range("D:D").Interior.Pattern = xlNone

	For i = 1 To 20

	x = Int((Rnd * 3) + 1)
	y = Int((Rnd * 3) + 1)
	z = Int((Rnd * 3) + 1)

	If x = 1 Then
		a = "A"
		ElseIf x = 2 Then
			a = "B"
			ElseIf x = 3 Then
				a = "C"
        
	End If

	If y = 1 Then
		b = "A"
		ElseIf y = 2 Then
			b = "B"
			ElseIf y = 3 Then
				b = "C"
        
	End If

	If z = 1 Then
		c = "A"
		ElseIf z = 2 Then
			c = "B"
			ElseIf z = 3 Then
				c = "C"
        
	End If

	Cells(i, 4) = a & b & c

	Next i

	For q = 1 To Cells(Rows.Count, "D").End(xlUp).Row
    
		If Cells(q, 4) Like "*A" Then
			Cells(q, 4).Interior.Color = vbRed
        
		End If
    
	Next q
    
	End Sub
