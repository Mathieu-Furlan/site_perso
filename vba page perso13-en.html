Thirteenth case: summing values by months.

Summary of the problem:

I need to sum data by month in one column. Month column contains dates.

Generate dates randomly:

	
	Sub CreerDates()

	Dim i As Integer
	Dim u As Integer
	Dim jour
	Dim mois
	Dim annee

	Randomize

	For i = 1 To 20

	jour = Int((Rnd * 31) + 1)
	mois = Int((Rnd * 3) + 1)

	Cells(i, 1) = jour & "/" & mois & "/" & 2008
	Cells(i, 2) = Int((Rnd * 7) + 1)

	Next i

	Columns(3).ClearContents
	Columns(4).ClearContents

	End Sub

Code to do what was asked:

	
	Sub SommeB()

	Dim q As Integer
	Dim i As Integer
	Dim u As Integer

	For q = 1 To 12
		For i = 1 To Cells(Rows.Count, 1).End(xlUp).Row
        
			If Month(Cells(i, 1)) = q Then
				u = u + Cells(i, 2)
				Cells(q, 3) = q
				Cells(q, 4) = u
				Debug.Print (u)
            
			ElseIf IsEmpty(Cells(q, 3)) Then
				Cells(q, 3) = q
				Cells(q, 4) = 0
            
			End If
    
		Next i
		u = 0
		Cells(q, 3) = MonthName(q) & " " & Year(Cells(q, 1))
	Next q

	End Sub
